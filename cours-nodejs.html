1 Introduction à Node.js

Node.js est un environnement d'exécution JavaScript côté serveur, construit sur le moteur V8 de Chrome. Il permet d'utiliser JavaScript pour le backend, créant ainsi un stack full JavaScript.

1.1 Pourquoi Node.js ?

  • Single-threaded avec Event Loop : Gère des milliers de connexions simultanées
  • NPM : Plus grand écosystème de packages au monde
  • Full-stack JavaScript : Même langage frontend et backend
  • Performances : Moteur V8 optimisé, I/O non bloquant
  • Microservices : Idéal pour les architectures modernes

1.2 Installation et premiers pas

Bash 
# Vérifier l'installation
node --version  # v20.x.x recommandé
npm --version

# Initialiser un projet
mkdir mon-api && cd mon-api
npm init -y

# Installer Express
npm install express

# Structure recommandée
mon-api/
├── src/

│   ├── middleware/     # Middlewares personnalisés
│   ├── utils/          # Fonctions utilitaires
│   └── app.js          # Point d'entrée
├── tests/
├── .env
├── .gitignore
└── package.json

1.3 Premier serveur Express

JavaScript - Serveur Express 
const express = require('express');
const app = express();
const PORT = process.env.PORT || 3000;

// Middleware pour parser le JSON
app.use(express.json());

// Route GET
app.get('/', (req, res) => {
  res.json({ message: 'Bienvenue sur mon API !' });
});

// Route avec paramètre
app.get('/users/:id', (req, res) => {
  const userId = req.params.id;
  res.json({ id: userId, name: 'Alice' });
});

// Route POST
app.post('/users', (req, res) => {
  const { name, email } = req.body;
  // Logique de création...
  res.status(201).json({ id: 1, name, email });
});

// Gestion des erreurs 404
app.use((req, res) => {
  res.status(404).json({ error: 'Route non trouvée' });
});

app.listen(PORT, () => {
  console.log(`Serveur démarré sur http://localhost:${PORT}`);
});

2 Routing et Middleware

2.1 Système de routing avancé

JavaScript - Routing Express 
const express = require('express');
const router = express.Router();

// Routes chaînées
router.route('/products')
  .get(getAllProducts)
  .post(createProduct);

router.route('/products/:id')
  .get(getProduct)
  .put(updateProduct)
  .patch(patchProduct)

// Route avec regex
app.get('/users/:id([0-9]+)', (req, res) => {
  res.json({ userId: req.params.id });
});

// Query parameters
app.get('/search', (req, res) => {
  const { q, category, sort } = req.query;
  // q=iphone&category=electronics&sort=price
  res.json({ query: q, category, sort });
});

// Middleware de route
app.get('/admin', isAuthenticated, isAdmin, (req, res) => {
  res.json({ message: 'Panel admin' });
});

2.2 Middlewares personnalisés

JavaScript - Middlewares 
// Logger personnalisé
const logger = (req, res, next) => {
  console.log(`${new Date().toISOString()} - ${req.method} ${req.url}`);
  next(); // Passe au middleware/route suivant
};
app.use(logger);

// Validation des données
const validateUser = (req, res, next) => {
  const { name, email } = req.body;

  if (!name || !email) {
    return res.status(400).json({ 
      error: 'Nom et email requis' 
    });
  }

  if (!email.includes('@')) {
    return res.status(400).json({ 
      error: 'Email invalide' 
    });
  }

  next();
};
app.post('/users', validateUser, createUser);

// Gestion des erreurs globale
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(err.status || 500).json({
    error: err.message || 'Erreur serveur interne'
  });
});

2.3 Middlewares tiers essentiels

  • cors : Gestion des requêtes cross-origin
  • helmet : Sécurisation des headers HTTP
  • morgan : Logger de requêtes HTTP
  • compression : Compression gzip des réponses
  • rate-limit : Limitation du nombre de requêtes

3 Base de données avec MongoDB

MongoDB est une base de données NoSQL orientée documents, parfaitement adaptée aux applications Node.js grâce à sa flexibilité et sa scalabilité.

3.1 Introduction à MongoDB et Mongoose

JavaScript - Mongoose 
const mongoose = require('mongoose');

// Connexion à MongoDB
mongoose.connect(process.env.MONGODB_URI)
  .then(() => console.log('Connecté à MongoDB'))
  .catch(err => console.error('Erreur de connexion:', err));

// Définition d'un schéma
const userSchema = new mongoose.Schema({
  name: { 
    type: String, 
    required: [true, 'Le nom est requis'],
    trim: true,
    maxlength: [50, 'Nom trop long']
  },
  email: {
    type: String,
    required: true,
    unique: true,
    lowercase: true,
    match: [/\S+@\S+\.\S+/, 'Email invalide']
  },
  age: {
    type: Number,
    min: [0, 'Âge positif requis'],
    max: [150, 'Âge invalide']
  },
  role: {
    type: String,
    enum: ['user', 'admin', 'moderator'],
    default: 'user'
  },
  createdAt: {
    type: Date,
    default: Date.now
  }
});

// Méthode d'instance
userSchema.methods.getFullInfo = function() {
  return `${this.name} (${this.email})`;
};

// Méthode statique
userSchema.statics.findByEmail = function(email) {
  return this.findOne({ email });
};

const User = mongoose.model('User', userSchema);

Sécurité

Ne jamais inclure directement des données utilisateur dans les requêtes MongoDB sans validation. Utilisez toujours Mongoose avec validation de schéma ou sanitize les entrées pour prévenir les injections NoSQL.

4 Authentification et Sécurité

4.1 JWT (JSON Web Tokens)

JWT est le standard moderne pour l'authentification stateless. Il permet de vérifier l'identité d'un utilisateur sans session serveur.

JavaScript - JWT Auth 
const jwt = require('jsonwebtoken');
const bcrypt = require('bcryptjs');

// Inscription
const register = async (req, res) => {
  const { name, email, password } = req.body;

  // Vérifier si l'email existe déjà
  const existingUser = await User.findOne({ email });
  if (existingUser) {
    return res.status(409).json({ error: 'Email déjà utilisé' });
  }

  // Hasher le mot de passe (coût 12)
  const hashedPassword = await bcrypt.hash(password, 12);

  const user = await User.create({
    name,
    email,
    password: hashedPassword
  });

  res.status(201).json({ 
    message: 'Utilisateur créé',
    user: { id: user._id, name: user.name, email: user.email }
  });
};

// Connexion
const login = async (req, res) => {
  const { email, password } = req.body;

  const user = await User.findOne({ email }).select('+password');
  if (!user) {
    return res.status(401).json({ error: 'Identifiants invalides' });
  }

  const isValid = await bcrypt.compare(password, user.password);
  if (!isValid) {
    return res.status(401).json({ error: 'Identifiants invalides' });
  }

  // Générer le JWT
  const token = jwt.sign(
    { userId: user._id, role: user.role },
    process.env.JWT_SECRET,
    { expiresIn: '7d' }
  );

  res.json({ token, user: { id: user._id, name: user.name } });
};

// Middleware d'authentification
const authenticate = async (req, res, next) => {
  try {
    const token = req.headers.authorization?.split(' ')[1];

    if (!token) {
      return res.status(401).json({ error: 'Token manquant' });
    }

    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    req.user = await User.findById(decoded.userId);
    next();
  } catch (error) {
    res.status(401).json({ error: 'Token invalide' });
  }
};

// Route protégée
app.get('/profile', authenticate, (req, res) => {
  res.json({ user: req.user });
});

4.2 Sécurité web essentielle

  • Helmet : Sécurise les headers HTTP (XSS, clickjacking, etc.)
  • Rate Limiting : Limite les requêtes par IP (100 req/15min)
  • Input Sanitization : Nettoie les entrées utilisateur
  • CORS : Configure les origines autorisées
  • HTTPS : Toujours en production
  • Variables d'environnement : Ne jamais committer les secrets

5 Architecture API REST complète

JavaScript - API REST complète 
// src/routes/productRoutes.js
const express = require('express');
const router = express.Router();
const productController = require('../controllers/productController');
const { authenticate, authorize } = require('../middleware/auth');

router
  .route('/')
  .get(productController.getAllProducts)
  .post(authenticate, authorize('admin'), productController.createProduct);

router
  .route('/:id')
  .get(productController.getProduct)
  .patch(authenticate, authorize('admin'), productController.updateProduct)
  .delete(authenticate, authorize('admin'), productController.deleteProduct);

// src/controllers/productController.js
const Product = require('../models/Product');

exports.getAllProducts = async (req, res, next) => {
  try {
    const { page = 1, limit = 10, sort, category, search } = req.query;

    const query = {};
    if (category) query.category = category;
    if (search) query.name = { $regex: search, $options: 'i' };

    const products = await Product.find(query)
      .sort(sort || '-createdAt')
      .limit(limit * 1)
      .skip((page - 1) * limit);

    const count = await Product.countDocuments(query);

    res.json({
      products,
      totalPages: Math.ceil(count / limit),
      currentPage: page,
      total: count
    });
  } catch (error) {
    next(error);
  }
};

exports.createProduct = async (req, res, next) => {
  try {
    const product = await Product.create(req.body);
    res.status(201).json(product);
  } catch (error) {
    next(error);
  }
};

CLOUDINARY_API_SECRET=xxx

6.2 Déploiement sur Render/Railway

  1. Créer un compte sur Render ou Railway
  2. Connecter votre repo GitHub
  3. Configurer les variables d'environnement
  4. Définir la commande de build : npm install
  5. Définir la commande de start : node src/app.js
  6. Déployer et obtenir l'URL de production

6.3 Bonnes pratiques production

  • Utiliser un process manager comme PM2 : pm2 start app.js
  • Activer la compression gzip
  • Configurer un reverse proxy (Nginx)
  • Mettre en place le logging structuré (Winston/Pino)
  • Monitorer avec des outils comme New Relic ou Datadog
