1 Introduction à Angular

Angular est un framework TypeScript développé par Google pour construire des applications web single-page (SPA). Contrairement à React ou Vue, Angular est un framework complet avec tout intégré.

1.1 Pourquoi Angular ?

  • Framework complet : Routing, forms, HTTP, testing inclus
  • TypeScript natif : Typage fort pour du code robuste
  • Dependency Injection : Architecture modulaire et testable
  • RxJS intégré : Programmation réactive native
  • Angular CLI : Outils de développement puissants
  • Enterprise-ready : Utilisé par Google, Microsoft, Forbes

1.2 Installation et Angular CLI

Bash 
# Installer Angular CLI globalement
npm install -g @angular/cli

# Créer un nouveau projet
ng new mon-app --routing --style=scss --standalone

# Options : --routing (routing), --style=scss, --standalone (sans NgModules)

cd mon-app
ng serve  # Démarre le serveur de développement

# Générer des composants
ng generate component mon-composant
ng g c mon-composant --standalone

# Générer un service
ng g service services/user

# Générer un module
ng g module features/products --routing

# Build production
ng build --configuration production

2.2 Classes et décorateurs

TypeScript - Classes 
// Classe avec access modifiers
class Product {
  // Propriétés
  private _price: number;
  public readonly id: string;

  // Constructor avec shorthand
  constructor(
    public name: string,
    public description: string,
    price: number,
    id: string
  ) {
    this._price = price;
    this.id = id;
  }

  // Getter
  get price(): number {
    return this._price;
  }

  // Setter
  set price(value: number) {
    if (value < 0) throw new Error('Prix invalide');
    this._price = value;
  }

  // Méthode
  getDiscountedPrice(discount: number): number {
    return this._price * (1 - discount);
  }
}

// Héritage
class DigitalProduct extends Product {
  constructor(
    name: string,
    description: string,
    price: number,
    id: string,
    public downloadUrl: string
  ) {
    super(name, description, price, id);
  }
}

3 Composants Angular

Les composants sont les blocs de construction d'une application Angular. Chaque composant contrôle une portion de l'écran appelée view.

3.1 Composant standalone (Angular 14+)

TypeScript - Composant Standalone 
import { Component, Input, Output, EventEmitter } from '@angular/core';
import { CommonModule } from '@angular/common';

@Component({
  selector: 'app-user-card',
  standalone: true,
  imports: [CommonModule],  // Modules/composants utilisés
  template: `
    <div class="card" [class.active]="isActive">
      <img [src]="user.avatar" [alt]="user.name">
      <h3>{{ user.name }}</h3>
      <p>{{ user.email }}</p>
      <button (click)="onSelect()">Sélectionner</button>
    </div>
  `,
  styles: [`
    .card { padding: 1rem; border: 1px solid #ddd; }
    .active { border-color: blue; }
    img { width: 50px; height: 50px; border-radius: 50%; }
  `]
})
export class UserCardComponent {
  @Input() user!: User;
  @Input() isActive = false;
  @Output() select = new EventEmitter<User>();

  onSelect() {
    this.select.emit(this.user);
  }
}

3.3 Signals (Angular 16+)

Les Signals sont le nouveau système de réactivité d'Angular, plus performant et prévisible que Zone.js :

TypeScript - Signals 
import { Component, signal, computed, effect } from '@angular/core';

@Component({
  selector: 'app-counter',
  standalone: true,
  template: `
    <p>Compteur : {{ count() }}</p>
    <p>Double : {{ doubleCount() }}</p>
    <button (click)="increment()">+</button>
    <button (click)="decrement()">-</button>
  `
})
export class CounterComponent {
  // Signal writable
  count = signal(0);

  // Signal computed (dérivé)
  doubleCount = computed(() => this.count() * 2);

  // Effect (side effect)
  constructor() {
    effect(() => {
      console.log('Le compteur a changé :', this.count());
    });
  }

  increment() {
    this.count.update(value => value + 1);
  }

  decrement() {
    this.count.set(this.count() - 1);
  }
}

4 Services et Dependency Injection

TypeScript - Service Angular 
import { Injectable, inject } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable, BehaviorSubject } from 'rxjs';

// Injectable avec providedIn: 'root' = singleton global
@Injectable({
  providedIn: 'root'
})
export class UserService {
  private http = inject(HttpClient);
  private apiUrl = 'https://api.example.com/users';

  // State management avec BehaviorSubject
  private users$ = new BehaviorSubject<User[]>([]);
  public users = this.users$.asObservable();

  getUsers(): Observable<User[]> {
    return this.http.get<User[]>(this.apiUrl);
  }

  getUser(id: number): Observable<User> {
    return this.http.get<User>(`${this.apiUrl}/${id}`);
  }

  createUser(user: Partial<User>): Observable<User> {
    return this.http.post<User>(this.apiUrl, user);
  }

  updateUser(id: number, user: Partial<User>): Observable<User> {
    return this.http.patch<User>(`${this.apiUrl}/${id}`, user);
  }

  deleteUser(id: number): Observable<void> {
    return this.http.delete<void>(`${this.apiUrl}/${id}`);
  }
}

// Utilisation dans un composant
@Component({...})
export class UserListComponent {
  private userService = inject(UserService);
  users$ = this.userService.getUsers();
}

4.1 Interceptors HTTP

TypeScript - Interceptor 
import { HttpInterceptorFn } from '@angular/common/http';

export const authInterceptor: HttpInterceptorFn = (req, next) => {
  const token = localStorage.getItem('token');

  if (token) {
    req = req.clone({
      setHeaders: { Authorization: `Bearer ${token}` }
    });
  }

  return next(req);
};

// Configuration dans app.config.ts
import { provideHttpClient, withInterceptors } from '@angular/common/http';

export const appConfig: ApplicationConfig = {
  providers: [
    provideHttpClient(withInterceptors([authInterceptor, errorInterceptor]))
  ]
};

5 RxJS - Programmation Réactive

RxJS est une bibliothèque de programmation réactive utilisée intensivement dans Angular pour gérer les flux de données asynchrones.

5.1 Observables et opérateurs essentiels

TypeScript - RxJS 
import { Observable, of, from, Subject, BehaviorSubject } from 'rxjs';
import { map, filter, switchMap, mergeMap, catchError, debounceTime, distinctUntilChanged, takeUntil } from 'rxjs/operators';

// Création d'observables
const numbers$ = of(1, 2, 3, 4, 5);
const users$ = from(fetchUsers());

// Subject (observable + observer)
const searchSubject = new Subject<string>();

// BehaviorSubject (avec valeur initiale)
const currentUser$ = new BehaviorSubject<User | null>(null);

// Opérateurs de transformation
searchSubject.pipe(
  debounceTime(300),           // Attendre 300ms après la dernière frappe
  distinctUntilChanged(),      // Ignorer si valeur identique
  switchMap(term =>            // Annuler la requête précédente
    this.http.get(`/api/search?q=${term}`).pipe(
      catchError(() => of([])) // Gérer les erreurs
    )
  )
).subscribe(results => {
  this.searchResults = results;
});

// Combinaison d'observables
import { combineLatest, forkJoin } from 'rxjs';

// combineLatest : émet quand n'importe quel observable émet
combineLatest([users$, posts$]).subscribe(([users, posts]) => {
  // Mise à jour de l'UI
});

// forkJoin : attend que tous les observables soient complets
forkJoin([getUser(), getOrders(), getStats()]).subscribe(
  ([user, orders, stats]) => {
    // Toutes les données sont chargées
  }
);

// Gestion du cycle de vie avec takeUntil
private destroy$ = new Subject<void>();

ngOnInit() {
  this.data$.pipe(
    takeUntil(this.destroy$)
  ).subscribe(data => {
    this.data = data;
  });
}

ngOnDestroy() {
  this.destroy$.next();
  this.destroy$.complete();
}
Bonne pratique RxJS

Toujours se désabonner des observables dans ngOnDestroy pour éviter les memory leaks. Utilisez takeUntil avec un Subject de destruction ou le pipe async dans le template.

6 Formulaires Réactifs

Angular propose deux approches pour les formulaires : Template-driven (simple) et Reactive (puissant et testable). Les formulaires réactifs sont recommandés pour les applications complexes.

TypeScript - Reactive Forms 
import { Component, inject } from '@angular/core';
import { FormBuilder, FormGroup, Validators, FormArray } from '@angular/forms';
import { ReactiveFormsModule } from '@angular/forms';

@Component({
  selector: 'app-user-form',
  standalone: true,
  imports: [ReactiveFormsModule, CommonModule],
  template: `
    <form [formGroup]="userForm" (ngSubmit)="onSubmit()">
      <div>
        <label>Nom</label>
        <input formControlName="name">
        <div *ngIf="name?.invalid && name?.touched">
          <span *ngIf="name?.errors?.['required']">Nom requis</span>
          <span *ngIf="name?.errors?.['minlength']">Min 3 caractères</span>
        </div>
      </div>

      <div>
        <label>Email</label>
        <input formControlName="email" type="email">
      </div>

      <div formGroupName="address">
        <label>Rue</label>
        <input formControlName="street">
        <label>Ville</label>
        <input formControlName="city">
      </div>

      <div formArrayName="skills">
        <div *ngFor="let skill of skills.controls; let i = index">
          <input [formControlName]="i">
          <button type="button" (click)="removeSkill(i)">Supprimer</button>
        </div>
        <button type="button" (click)="addSkill()">Ajouter une compétence</button>
      </div>

      <button type="submit" [disabled]="userForm.invalid">Soumettre</button>
    </form>
  `
})
export class UserFormComponent {
  private fb = inject(FormBuilder);

  userForm = this.fb.group({
    name: ['', [Validators.required, Validators.minLength(3)]],
    email: ['', [Validators.required, Validators.email]],
    address: this.fb.group({
      street: ['', Validators.required],
      city: ['', Validators.required]
    }),
    skills: this.fb.array([this.fb.control('')])
  });

  get name() { return this.userForm.get('name'); }
  get skills() { return this.userForm.get('skills') as FormArray; }

  addSkill() {
    this.skills.push(this.fb.control(''));
  }

  removeSkill(index: number) {
    this.skills.removeAt(index);
  }

  onSubmit() {
    if (this.userForm.valid) {
      console.log(this.userForm.value);
    }
  }
}

7 Routing et Navigation

TypeScript - Routing Angular 
// app.routes.ts
import { Routes } from '@angular/router';
import { authGuard } from './core/guards/auth.guard';

export const routes: Routes = [
  { 
    path: '', 
    loadComponent: () => import('./features/home/home.component')
      .then(m => m.HomeComponent) 
  },
  { 
    path: 'products',
    loadChildren: () => import('./features/products/products.routes')
      .then(m => m.PRODUCTS_ROUTES)
  },
  { 
    path: 'profile',
    loadComponent: () => import('./features/profile/profile.component')
      .then(m => m.ProfileComponent),
    canActivate: [authGuard]
  },
  { path: '**', redirectTo: '' }
];

// Guard fonctionnel (Angular 15+)
export const authGuard: CanActivateFn = () => {
  const authService = inject(AuthService);
  const router = inject(Router);

  if (authService.isAuthenticated()) {
    return true;
  }

  return router.createUrlTree(['/login']);
};

// Navigation programmatique
@Component({...})
export class ProductComponent {
  private router = inject(Router);
  private route = inject(ActivatedRoute);

  goToDetails(id: number) {
    this.router.navigate(['/products', id]);
    // ou
    this.router.navigate(['../details', id], { relativeTo: this.route });
  }
}
